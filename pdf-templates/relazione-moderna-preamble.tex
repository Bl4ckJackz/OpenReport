% ======================================================================
% Relazione moderna colorata — preamble LaTeX
% ----------------------------------------------------------------------
% Da includere all'inizio di RELAZIONE.tex (dopo \documentclass) per
% ottenere un PDF "relazione aziendale" vivace invece del classico
% accademico B/N.
%
% Palette:
%   primary  #1E40AF   blu profondo       → sezioni, copertina, link
%   accent   #F59E0B   ambra              → sottotitoli, regola, URL
%   muted    #64748B   grigio bluastro    → metadata, footer, caption
%   soft     #EFF6FF   azzurro chiarissimo → sfondo callout/tabelle
%
% Personalizza i \definecolor sotto per adattare alla tua palette.
%
% Compilazione consigliata:
%   xelatex --shell-escape RELAZIONE.tex   (shell-escape per minted)
%   biber RELAZIONE                         (se biblatex)
%   xelatex --shell-escape RELAZIONE.tex
%   xelatex --shell-escape RELAZIONE.tex
% ======================================================================

% -------------------- Encoding & font (xelatex) ------------------------
\usepackage{fontspec}
\defaultfontfeatures{Ligatures=TeX}
\setmainfont{Calibri}
\setsansfont{Calibri}
\setmonofont{Consolas}

\usepackage[italian]{babel}
\usepackage{csquotes}

% -------------------- Geometria & interlinea ---------------------------
\usepackage[a4paper,margin=2.3cm,top=2.8cm,bottom=2.8cm]{geometry}
\usepackage{setspace}
\setstretch{1.3}

% -------------------- Palette ------------------------------------------
\usepackage{xcolor}
\definecolor{primary}{HTML}{1E40AF}
\definecolor{accent}{HTML}{F59E0B}
\definecolor{muted}{HTML}{64748B}
\definecolor{soft}{HTML}{EFF6FF}
\definecolor{softaccent}{HTML}{FEF3C7}

% -------------------- Link colorati ------------------------------------
\usepackage{hyperref}
\hypersetup{
  colorlinks=true,
  linkcolor=primary,
  urlcolor=accent,
  citecolor=primary,
  pdfborder={0 0 0}
}

% -------------------- Titoli colorati (titlesec) -----------------------
\usepackage{titlesec}
\titleformat{\section}
  {\Large\bfseries\color{primary}}
  {\thesection}{0.6em}{}
  [\vspace{-0.5em}{\color{accent}\rule{\linewidth}{0.6pt}}]
\titleformat{\subsection}
  {\large\bfseries\color{primary}}
  {\thesubsection}{0.6em}{}
\titleformat{\subsubsection}
  {\normalsize\bfseries\color{accent}}
  {\thesubsubsection}{0.6em}{}

\titlespacing*{\section}{0pt}{1.4em}{0.6em}
\titlespacing*{\subsection}{0pt}{1.1em}{0.5em}

% -------------------- Header / footer (fancyhdr) -----------------------
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0pt}
\fancyhead[L]{\small\color{muted}\nouppercase{\leftmark}}
\fancyhead[R]{\small\color{muted}\thepage}
\fancyfoot[C]{\small\color{muted}\@title}
\makeatletter
\let\oldheadrule\headrule
\renewcommand{\headrule}{\color{accent}\oldheadrule}
\makeatother

% -------------------- Callout box (tcolorbox) --------------------------
\usepackage[most]{tcolorbox}
\newtcolorbox{nota}[1][]{%
  colback=soft, colframe=primary, boxrule=0.8pt, arc=2mm,
  left=4mm, right=4mm, top=2mm, bottom=2mm,
  title=#1, fonttitle=\bfseries\color{white}
}
\newtcolorbox{warn}[1][]{%
  colback=softaccent, colframe=accent, boxrule=1pt, arc=2mm,
  left=4mm, right=4mm, top=2mm, bottom=2mm,
  title=#1, fonttitle=\bfseries\color{black}
}
\newtcolorbox{info}[1][]{%
  colback=white, colframe=muted, boxrule=0.6pt, arc=1mm,
  left=3mm, right=3mm, top=1.5mm, bottom=1.5mm,
  title=#1, fonttitle=\bfseries\color{muted}
}

% -------------------- Tabelle a righe alternate ------------------------
\usepackage{array}
\usepackage{booktabs}
\usepackage{colortbl}
\usepackage{longtable}
\arrayrulecolor{muted}
\renewcommand{\arraystretch}{1.15}
% Attivare \rowcolors{2}{soft}{white} prima della tabella quando serve

% -------------------- Code con minted (colorato) -----------------------
% Richiede compilazione con --shell-escape e Pygments installato
\usepackage{minted}
\setminted{
  bgcolor=soft,
  fontsize=\small,
  frame=leftline,
  framesep=2mm,
  rulecolor=primary,
  linenos=false,
  breaklines=true,
  breakanywhere=true
}

% -------------------- Figure & caption ---------------------------------
\usepackage{graphicx}
\usepackage[font=small,labelfont={bf,color=primary},textfont=it]{caption}

% -------------------- Liste personalizzate -----------------------------
\usepackage{enumitem}
\setlist[itemize,1]{label=\textcolor{accent}{\textbullet},leftmargin=*}
\setlist[itemize,2]{label=\textcolor{primary}{$\circ$}}

% -------------------- Copertina (\maketitle rimpiazzato) ---------------
\makeatletter
\renewcommand{\maketitle}{%
  \begin{titlepage}
    \pagecolor{primary}\color{white}
    \vspace*{2cm}
    \noindent{\color{accent}\rule{\linewidth}{4pt}}\\[1.2em]
    {\Huge\bfseries \@title\par}
    \vspace{0.5em}
    \noindent{\color{accent}\rule{0.35\linewidth}{1.5pt}}\\[2em]
    {\Large\@author\par}
    \vfill
    {\large \@date\par}
    \vspace{1cm}
    \nopagecolor
  \end{titlepage}
}
\makeatother
